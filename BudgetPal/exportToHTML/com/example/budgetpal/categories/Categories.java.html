102      public void confirmDelete(int position) {
103          AlertDialog alertDialog = new AlertDialog.Builder(this)
104                  .setIcon(android.R.drawable.ic_dialog_alert)
105                  .setTitle("Are you sure to Delete?")
106                  .setMessage("Record " + position + " will be deleted permanently.")
107                  .setPositiveButton("Yes", new DialogInterface.OnClickListener() {
108                      @Override
109                      public void onClick(DialogInterface dialogInterface, int i) {
110                          DataModel modal = recordsList.get(position);
111                          dbConn.deleteCategory(String.valueOf(modal.getCatId()));
112  
113                          recordsList.remove(position);
114                          recyclerAdapterCat.notifyItemRemoved(position);
115                      }
116                  })
117                  .setNegativeButton("No", new DialogInterface.OnClickListener() {
118                      @Override
119                      public void onClick(DialogInterface dialogInterface, int i) {
120                          recreate();
121                      }
122                  })
123                  .show();
124      }

129      ItemTouchHelper.SimpleCallback simpleCallback = new ItemTouchHelper.SimpleCallback(0, ItemTouchHelper.LEFT | ItemTouchHelper.RIGHT) {
130          @Override
131          public boolean onMove(@NonNull RecyclerView recyclerView, @NonNull RecyclerView.ViewHolder viewHolder, @NonNull RecyclerView.ViewHolder target) {
132              return false;
133          }
134  
135          @Override
136          public void onSwiped(@NonNull RecyclerView.ViewHolder viewHolder, int direction) {
137  
138              int position = viewHolder.getAdapterPosition();
139  
140              switch (direction) {
141                  case ItemTouchHelper.LEFT:
142                      confirmDelete(position);
143                      break;
144  
145                  case ItemTouchHelper.RIGHT:
146                      EditCategoryDialog(position);
147                      break;
148              }
149  
150          }
151  
152          @Override
153          public void onChildDraw(@NonNull Canvas c, @NonNull RecyclerView recyclerView, @NonNull RecyclerView.ViewHolder viewHolder, float dX, float dY, int actionState, boolean isCurrentlyActive) {
154              new RecyclerViewSwipeDecorator.Builder(Categories.this, c, recyclerView, viewHolder, dX, dY, actionState, isCurrentlyActive)
155                      .addSwipeLeftBackgroundColor(ContextCompat.getColor(Categories.this, R.color.red))
156                      .addSwipeLeftActionIcon(R.drawable.ic_delete_white_24dp)
157                      .addSwipeRightBackgroundColor(ContextCompat.getColor(Categories.this, R.color.green))
158                      .addSwipeRightActionIcon(R.drawable.ic_edit_white_24dp)
159                      .create()
160                      .decorate();
161  
162              super.onChildDraw(c, recyclerView, viewHolder, dX, dY, actionState, isCurrentlyActive);
163          }

168      public void addNewCategoryDialog() {
169          dialogBuilder = new AlertDialog.Builder(this);
170          final View addCategoryPopupView = getLayoutInflater().inflate(R.layout.add_category, null);
171  
172          AddCatName = (EditText) addCategoryPopupView.findViewById(R.id.AddCatName);
173          AddCatDescription = (EditText) addCategoryPopupView.findViewById(R.id.AddCatDescription);
174          radioAddCatCreditType = (RadioButton) addCategoryPopupView.findViewById(R.id.radioAddCatCreditType);
175          radioAddCatDebitType = (RadioButton) addCategoryPopupView.findViewById(R.id.radioAddCatDebitType);
176          AddCatBudget = (EditText) addCategoryPopupView.findViewById(R.id.AddCatBudget);
177          btnAddCatSave = (Button) addCategoryPopupView.findViewById(R.id.btnAddCatSave);
178          btnAddCatCancel = (Button) addCategoryPopupView.findViewById(R.id.btnAddCatCancel);
179  
180          radioAddCatCreditType.setOnClickListener(new View.OnClickListener() {
181              @Override
182              public void onClick(View v) {
183                  boolean checked = ((RadioButton) v).isChecked();
184                  if (checked) {
185                      catType = "Credit";
186                      AddCatBudget.setVisibility(View.INVISIBLE);
187                      AddCatBudget.setText("0.0");
188                  }
189              }
190          });
191  
192          radioAddCatDebitType.setOnClickListener(new View.OnClickListener() {
193              @Override
194              public void onClick(View v) {
195                  boolean checked = ((RadioButton) v).isChecked();
196                  if (checked) {
197                      catType = "Debit";
198                      AddCatBudget.setVisibility(View.VISIBLE);
199  
200                      double leftBudgetForNewCategory = dbConn.readLeftBudgetForNewCategory();
201  
202                      AddCatBudget.setHint("Budget (" + leftBudgetForNewCategory + " left to allocate)");
203                  }
204              }
205          });
206  
207          dialogBuilder.setView(addCategoryPopupView);
208          dialog = dialogBuilder.create();
209          dialog.show();
210  
211          btnAddCatSave.setOnClickListener(new View.OnClickListener() {
212              @Override
213              public void onClick(View v) {
214                  if (TextUtils.isEmpty(AddCatName.getText())) {
215                      AddCatName.setError("Mandatory field - Enter Category name");
216                  } else if (TextUtils.isEmpty(AddCatDescription.getText())) {
217                      AddCatDescription.setError("Mandatory field - Enter Category description");
218                  } else if (!(radioAddCatCreditType.isChecked() || radioAddCatDebitType.isChecked())) {
219                      radioAddCatCreditType.setError("Mandatory field - Select Credit/Debit");
220                      radioAddCatDebitType.setError("Mandatory field - Select Credit/Debit");
221                  } else {
222                      recreate();
223                      dbConn.addNewCategory(AddCatName.getText().toString(), AddCatDescription.getText().toString(), String.valueOf(catType), Double.parseDouble(AddCatBudget.getText().toString()));
224                  }
225              }
226          });
227  
228          btnAddCatCancel.setOnClickListener(new View.OnClickListener() {
229              @Override
230              public void onClick(View v) {
231                  //Cancel code
232                  dialog.dismiss();
233              }
234          });
235      }
