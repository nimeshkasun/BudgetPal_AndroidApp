250          EditTransCategory = (Spinner) editTransactionPopupView.findViewById(R.id.EditTransCategory);

265          dbConn = new DbConn(Transactions.this);

271              categoryNames.add(modalCatNames.getCatName());
272          }
273          ArrayAdapter<String> categoryNamesAdapter = new ArrayAdapter<>(this, android.R.layout.simple_spinner_item, categoryNames);
274          categoryNamesAdapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
275          EditTransCategory.setAdapter(categoryNamesAdapter);
276  
277          for (int i = 0; i < recordsListCatNames.size(); i++) {
278              DataModel modalCatNames = recordsListCatNames.get(i);

280                  EditTransCategory.setSelection(i);

282          }

284  
285          dialogBuilder.setView(editTransactionPopupView);
286          dialog = dialogBuilder.create();
287          dialog.show();
288  
289          btnEditTransSave.setOnClickListener(new View.OnClickListener() {
290              @Override
291              public void onClick(View v) {
292                  if (TextUtils.isEmpty(EditTransDescription.getText())) {
293                      EditTransDescription.setError("Mandatory field - Enter Transaction description");
294                  } else if (TextUtils.isEmpty(EditTransPrice.getText())) {
295                      EditTransPrice.setError("Mandatory field - Enter Transaction price");
296                  } else {
297                      recreate();
298                      //Save code
299                      DataModel modal = recordsList.get(position);
300                      dbConn.updateTransaction(modal.getTranId(), EditTransCategory.getSelectedItem().toString(), String.valueOf(EditTransDate.getText()), EditTransDescription.getText().toString(), String.valueOf(EditTransRecurring.isChecked()), Double.parseDouble(EditTransPrice.getText().toString()));
301                  }

303          });

305          btnEditTransCancel.setOnClickListener(new View.OnClickListener() {
306              @Override
307              public void onClick(View v) {
308                  //Cancel code
309                  dialog.dismiss();
310                  recreate();
311              }
312          });

316      private void initDatePicker() {
317          DatePickerDialog.OnDateSetListener dateSetListener = new DatePickerDialog.OnDateSetListener() {
318              @Override
319              public void onDateSet(DatePicker datePicker, int year, int month, int day) {
320                  month = month + 1;
321                  String date = makeDateString(day, month, year);
322                  try {
323                      AddTransDate.setText(date);
324                  } catch (Exception e) {
325                      EditTransDate.setText(date);
326                  }
327              }
328          };
329  
330          Calendar cal = Calendar.getInstance();
331          int year = cal.get(Calendar.YEAR);
332          int month = cal.get(Calendar.MONTH);
333          int day = cal.get(Calendar.DAY_OF_MONTH);
334  
335          int style = android.R.style.Theme_Holo_Light_Dialog;
336  
337          datePickerDialog = new DatePickerDialog(this, style, dateSetListener, year, month, day);
338      }

340      private String getTodaysDate() {
341          Calendar cal = Calendar.getInstance();
342          int year = cal.get(Calendar.YEAR);
343          int month = cal.get(Calendar.MONTH);
344          month = month + 1;
345          int day = cal.get(Calendar.DAY_OF_MONTH);
346          return makeDateString(day, month, year);
347      }
348  
349      private String makeDateString(int day, int month, int year) {
350          return getMonthFormat(month) + " " + day + " " + year;
351      }
352  
353      private String getMonthFormat(int month) {
354          if (month == 1)
355              return "JAN";
356          if (month == 2)
357              return "FEB";
358          if (month == 3)
359              return "MAR";
360          if (month == 4)
361              return "APR";
362          if (month == 5)
363              return "MAY";
364          if (month == 6)
365              return "JUN";
366          if (month == 7)
367              return "JUL";
368          if (month == 8)
369              return "AUG";
370          if (month == 9)
371              return "SEP";
372          if (month == 10)
373              return "OCT";
374          if (month == 11)
375              return "NOV";
376          if (month == 12)
377              return "DEC";
378  
379          return "JAN";
380      }
