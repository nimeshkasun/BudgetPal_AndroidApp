172      public void updateTotalCalculations(String type, double transPrice) {
173  
174          SQLiteDatabase db = this.getWritableDatabase();
175  
176          Cursor cursorCalculations = db.rawQuery("SELECT totCredit AS Credit, totDebit AS Debit, totBalance AS Balance FROM " + totTABLE_NAME + " WHERE totId = '" + 1 + "'", null);
177          ArrayList<DataModel> categoryModalArrayList = new ArrayList<>();
178          double totCredit = 0.0;
179          double totDebit = 0.0;
180          double totBalance = 0.0;
181          if (cursorCalculations.moveToFirst()) {
182              do {
183                  totCredit = cursorCalculations.getDouble(cursorCalculations.getColumnIndex("Credit"));
184                  totDebit = cursorCalculations.getDouble(cursorCalculations.getColumnIndex("Debit"));
185                  totBalance = cursorCalculations.getDouble(cursorCalculations.getColumnIndex("Balance"));
186              } while (cursorCalculations.moveToNext());
187  
188          }
189          cursorCalculations.close();
190  
191          ContentValues values = new ContentValues();
192  
193          if (type.equals("Credit")) {
194              totCredit = totCredit + transPrice;
195              totBalance = totBalance + transPrice;
196          } else if (type.equals("Debit")) {
197              totDebit = totDebit + transPrice;
198              totBalance = totBalance - transPrice;
199          }
200  
201          values.put(totCREDIT_COL, totCredit);
202          values.put(totDEBIT_COL, totDebit);
203          values.put(totBALANCE_COL, totBalance);
204  
205          try {
206              if (db.update(totTABLE_NAME, values, "totId=?", new String[]{String.valueOf(1)}) == 0) {
207                  db.insert(totTABLE_NAME, null, values);
208                  Log.i("TOTCAL_Inserted", totBalance + "--" + totBalance + "--" + transPrice);
209              }
210          } catch (Exception e) {
211              db.insert(totTABLE_NAME, null, values);
212          }
213  
214          db.close();
215      }

376  
377          Cursor cursorTransactionsCredit = db.rawQuery("SELECT SUM(t.tPrice) as Sum FROM " + tTABLE_NAME + " t, " + cTABLE_NAME + " c WHERE t.tCategory = c.cName AND c.cType = 'Credit'", null);
378          Cursor cursorTransactionsDebit = db.rawQuery("SELECT SUM(cBudget) as Sum FROM " + cTABLE_NAME + " WHERE cType = 'Debit'", null);
379  
380          ArrayList<DataModel> transactionModalArrayList = new ArrayList<>();
381  
382          double credit = 0.0;
383          double debit = 0.0;
384          double left = 0.0;
385  
386          if (cursorTransactionsCredit.moveToFirst()) {
387              do {
388                  credit = cursorTransactionsCredit.getDouble(cursorTransactionsCredit.getColumnIndex("Sum"));
389              } while (cursorTransactionsCredit.moveToNext());
390          }
391  
392          if (cursorTransactionsDebit.moveToFirst()) {
393              do {
394                  debit = cursorTransactionsDebit.getDouble(cursorTransactionsDebit.getColumnIndex("Sum"));
395              } while (cursorTransactionsDebit.moveToNext());
396          }
397  
398          left = credit - debit;
399          Log.i("LEFTBUDGEST", left + "--" + credit + "--" + debit);
400  
401          cursorTransactionsCredit.close();
402
